### Task 1: Script skeleton — usage, argument parsing, validation

**Files:**

- [ ] **Step 1: Create the script with shebang, usage function, and argument parser**

```sh
#!/bin/sh
set -eu

usage() {
  cat <<'USAGE'
Usage: /app/admin.sh [OPTIONS] <COMMAND>

Options:
  --password <val>       Admin password (or set ADMIN_PASSWORD env var)
  --chain <name>         Chain name: ethereum, sepolia, bsc, bsc-testnet
  --namespace <val>      Namespace identifier (required)
  --endpoint <url>       Admin endpoint (default: OP_SIGNER_ADMIN_ENDPOINT or https://localhost:9545)
  --dry-run              Print commands without executing
  --continue-on-error    Continue after command failure

Commands:
  add-sequencer          Add sequencer config
  add-batcher            Add batcher config
  add-proposer           Add proposer config
  add-challenger-resolver  Add challenger-resolver config
  add-main               Add batcher + proposer + challenger-resolver
  add-all                Add all four components
  remove-sequencer       Remove sequencer config
  remove-batcher         Remove batcher config
  remove-proposer        Remove proposer config
  remove-challenger-resolver  Remove challenger-resolver config
  remove-all             Remove all four components
USAGE
  exit 1
}

# Defaults
PASSWORD="${ADMIN_PASSWORD:-}"
CHAIN=""
NAMESPACE=""
ENDPOINT="${OP_SIGNER_ADMIN_ENDPOINT:-https://localhost:9545}"
DRY_RUN=false
CONTINUE_ON_ERROR=false
COMMAND=""

# Parse arguments
while [ $# -gt 0 ]; do
  case "$1" in
    --password)
      PASSWORD="$2"; shift 2 ;;
    --chain)
      CHAIN="$2"; shift 2 ;;
    --namespace)
      NAMESPACE="$2"; shift 2 ;;
    --endpoint)
      ENDPOINT="$2"; shift 2 ;;
    --dry-run)
      DRY_RUN=true; shift ;;
    --continue-on-error)
      CONTINUE_ON_ERROR=true; shift ;;
    -h|--help)
      usage ;;
    -*)
      echo "Error: unknown option '$1'" >&2; usage ;;
    *)
      COMMAND="$1"; shift ;;
  esac
done

# Validate
if [ -z "$NAMESPACE" ]; then
  echo "Error: --namespace is required" >&2
  exit 1
fi

if [ -z "$PASSWORD" ]; then
  echo "Error: --password or ADMIN_PASSWORD env var is required" >&2
  exit 1
fi

if [ -z "$COMMAND" ]; then
  echo "Error: command is required" >&2
  usage
fi
```

- [ ] **Step 2: Add chain validation and mapping**

```sh
# Chain ID mapping
resolve_chain_id() {
  case "$1" in
    ethereum)    echo "1" ;;
    sepolia)     echo "11155111" ;;
    bsc)         echo "56" ;;
    bsc-testnet) echo "97" ;;
    *)           echo "Error: invalid chain '$1'. Must be: ethereum, sepolia, bsc, bsc-testnet" >&2; exit 1 ;;
  esac
}

# Validate chain for add-* commands
case "$COMMAND" in
  add-*)
    if [ -z "$CHAIN" ]; then
      echo "Error: --chain is required for add commands" >&2
      exit 1
    fi
    CHAIN_ID=$(resolve_chain_id "$CHAIN")
    ;;
  remove-*)
    CHAIN_ID=""
    ;;
  *)
    echo "Error: unknown command '$COMMAND'" >&2
    usage
    ;;
esac
```

```sh
# Component definitions: cn_suffix:key_name
COMP_SEQUENCER="sequencer:sequencer-prikey"
COMP_BATCHER="batcher:batcher-prikey"
COMP_PROPOSER="proposer:proposer-prikey"
COMP_CHALLENGER="challenger:challenger-resolver-prikey"

# Generate a single add-config command string
gen_add_cmd() {
  _cn_suffix="${1%%:*}"
  _key_name="${1##*:}"
  echo "/app/op-signer admin --admin.endpoint $ENDPOINT --admin.password $PASSWORD add-config --parent-chain-id $CHAIN_ID --allowed-client-cn \"${NAMESPACE}-seq-${_cn_suffix}\" --path \"${NAMESPACE}-operator-keys/${_key_name}\""
}

# Generate a single remove-config command string
gen_remove_cmd() {
  _key_name="${1##*:}"
  echo "/app/op-signer admin --admin.endpoint $ENDPOINT --admin.password $PASSWORD remove-config-by-path --path \"${NAMESPACE}-operator-keys/${_key_name}\""
}

# Generate a display label for a component command
gen_label() {
  _action="$1"
  _cn_suffix="${2%%:*}"
  echo "${_action} ${_cn_suffix}"
}
```

### Task 3: Preview, confirmation, and execution loop

**Files:**
- Modify: `op-signer/admin.sh`

- [ ] **Step 1: Add the preview function**

Append after the command list assembly:

```sh
# Print command preview with password masked
print_preview() {
  echo ""
  echo "=== Commands to execute ==="
  echo ""
  _i=1
  while [ "$_i" -le "$CMD_COUNT" ]; do
    eval "_cmd=\"\$CMD_${_i}\""
    _masked=$(echo "$_cmd" | sed "s|--admin.password $PASSWORD|--admin.password ****|g")
    echo "[$_i] $_masked"
    echo ""
    _i=$((_i + 1))
  done
}
```

- [ ] **Step 2: Add confirmation and execution loop**

```sh
# Print preview
print_preview

# Dry run — exit after preview
if [ "$DRY_RUN" = true ]; then
  echo "(dry-run mode, not executing)"
  exit 0
fi

# Confirm
printf "Proceed? [y/N] "
read -r _answer
case "$_answer" in
  y|Y) ;;
  *)   echo "Aborted."; exit 0 ;;
esac

# Execute
_succeeded=0
_failed=0
_skipped=0
_i=1
while [ "$_i" -le "$CMD_COUNT" ]; do
  eval "_cmd=\"\$CMD_${_i}\""
  eval "_label=\"\$LABEL_${_i}\""

  # Skip if prior failure and not continue-on-error
  if [ "$_failed" -gt 0 ] && [ "$CONTINUE_ON_ERROR" = false ]; then
    eval "RESULT_${_i}=SKIP"
    _skipped=$((_skipped + 1))
    _i=$((_i + 1))
    continue
  fi

  echo ""
  echo "--- [$_i] $_label ---"
  eval "_exit_code=0"
  eval "$_cmd" || _exit_code=$?

  if [ "$_exit_code" -eq 0 ]; then
    eval "RESULT_${_i}=OK"
    _succeeded=$((_succeeded + 1))
  else
    eval "RESULT_${_i}=\"FAIL:${_exit_code}\""
    _failed=$((_failed + 1))
  fi

  _i=$((_i + 1))
done
```

- [ ] **Step 3: Add result summary**

```sh
# Summary
echo ""
echo "=== Summary ==="
_i=1
while [ "$_i" -le "$CMD_COUNT" ]; do
  eval "_label=\"\$LABEL_${_i}\""
  eval "_result=\"\$RESULT_${_i}\""
  case "$_result" in
    OK)
      echo "[$_i] [OK]   $_label"
      ;;
    SKIP)
      echo "[$_i] [SKIP] $_label"
      ;;
    FAIL:*)
      _code="${_result#FAIL:}"
      echo "[$_i] [FAIL] $_label (exit code $_code)"
      ;;
  esac
  _i=$((_i + 1))
done

_total=$CMD_COUNT
echo ""
echo "Result: ${_succeeded}/${_total} succeeded, ${_failed} failed, ${_skipped} skipped"

# Exit with failure if any command failed
if [ "$_failed" -gt 0 ]; then
  exit 1
fi
```
